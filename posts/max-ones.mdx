## Solution

```rust
pub fn max_consecutive_ones(nums: Vec<i32>) -> i32 {
  let mut prev: i32 = 0;
  let mut curr: i32 = 0;
  let mid = (nums.len() / 2) as i32;

  for num in nums {
    if num == 1 {
      curr += 1;
    } else {
      if prev > mid  {
        return prev;
      }

      if prev < curr {
        if curr > mid {
          return curr;
        }
        prev = curr;
      } 
      curr = 0;
    }
  }

  if prev < curr {
    return curr;
  } else {
    return prev;
  }
}
```
