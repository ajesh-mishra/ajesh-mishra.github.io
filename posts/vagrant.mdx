```bash
# Check the version
vagrant --version

# Create a new VM
vagrant box add bento/centos-7.2 --provider=virtualbox

# Create a Vagrantfile to start the VM
vagrant init bento/centos-7.2

# Start the VM
vagrant up

# Shows status of all the VMs
vagrant status

# Login into the VM
vagrant ssh <default>
```
