## Solution

```rust
fn two_sum(nums: Vec<i32>, target: i32) -> Option<Vec<usize>> {
  for (i, num) in nums.iter().enumerate() {
    let mut j = i + 1;
    while j < nums.len() {
      if num + nums[j] == target {
        return Some(vec![i, j]);
      }
      j += 1;
    }
  }
  
  None
}

fn main() {
  if let Some(result) = two_sum(vec![2,8,11,15], 9) {
    println!("{:?}", result);
  } else {
    println!("No result!");
  }
}
```
