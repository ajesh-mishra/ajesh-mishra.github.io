## Solution

An integer is a palindrome when it reads the same backward as forward. For example, 121 is a palindrome while 123 is not.

```rust
fn is_palindrome(x: i32) -> bool {
  if x < 0 {
    return false;
  }

  let given_str = x.to_string();
  let mut rev_str = String::from("");

  for c in given_str.chars() {
    rev_str = format!("{}{}", c, rev_str);
  }

  if given_str == rev_str {
    return true;
  }

  false
}
```

    assert_eq!(is_palindrome(121), true);

    assert_eq!(is_palindrome(-121), false);
  }

  fn tc_3() {
    assert_eq!(is_palindrome(10), false);
