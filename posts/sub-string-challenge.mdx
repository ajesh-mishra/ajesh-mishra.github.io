## Solution

```rust
fn str_str(haystack: String, needle: String) -> i32 {
  let mut i = 0;
  let needle_len = needle.len();
  
  while (i + needle_len) <= haystack.len() {
    if haystack[i..i+needle_len] == needle {
      return i as i32;
    }
    i += 1;
  }
  
  return -1;
}
```
