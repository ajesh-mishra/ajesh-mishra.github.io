## Solution

```rust
fn main() {
  let score = 256;
  let mut count = 0;

  while count < 30 {
    if score == 2_i32.pow(count) {
      println!("{}", count);
      break;
    } else if score < 2_i32.pow(count) {
      for i in 0..count {
        for j in (i + 1)..count {
          if score == (2_i32.pow(i) + 2_i32.pow(j)) {
            println!("{} and {}", i, j);
          }
        }
      }
      break;
    }

    count += 1;
  }
}
```
